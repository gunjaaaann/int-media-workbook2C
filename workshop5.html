SELF PORTRAIT

Made with ThreeJS


                function setup() {
                    createCanvas(windowWidth, windowHeight);
                    background(255);
                }

                function draw() {
                    fill(255, 100, 10);
                    no stroke();

                    //face
                    circle(windowWidth/2, windowHeight/2, 500);

                    //hair
                    fill(180, 255, 255);
                    rect(920, 200, 100, 550);
                    fill(100, 200, 200);
                    rect(420, 200, 100, 550);
                    rect(420, 200, 600, 150);

                    //eyes
                    fill(0);
                    circle(600, 320, 40);
                    circle(850, 320, 40);

                    //mouth
                    fill(120, 0, 100);
                    stroke(0);
                    arc(windowWidth/2, windowHeight/2 + 50, 200, 200, 0, PI);
                }
