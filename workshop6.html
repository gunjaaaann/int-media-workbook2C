                    let img;
                    let colors = [];
                    let dotCount = 3000; 
                    
                    function preload() {
                      img = loadImage("data/w6.8.jpg");
                    }
                    
                    function setup() {
                      createCanvas(800, 600);
                      imageMode(CENTER);
                      noStroke();
                      img.resize(width, height - 100); 
                      
                      for (let i = 0; i < dotCount; i++) {
                        let x = floor(random(img.width));
                        let y = floor(random(img.height));
                        let col = img.get(x, y);
                        fill(col);
                        ellipse(x, y, 4, 4);
                        colors.push(col);
                      }
                    
                      let palette = getDominantColors(colors, 300);
                      drawBarcode(palette);
                    }
                    
                    function getDominantColors(colArray, numColors) {
                      let freq = {};
                      
                      for (let col of colArray) {
                        let key = col.toString();
                        freq[key] = (freq[key] || 0) + 1;
                      }
                    
                      return Object.entries(freq)
                        .sort((a, b) => b[1] - a[1])
                        .slice(0, numColors)
                        .map(entry => {
                          let [r, g, b, a] = entry[0].split(',').map(Number);
                          return color(r, g, b, a);
                        });
                    }
                    
                    function drawBarcode(palette) {
                      let barWidth = width / palette.length;
                      for (let i = 0; i < palette.length; i++) {
                        fill(palette[i]);
                        rect(i * barWidth, height - 100, barWidth, 100);
                      }
                    }
